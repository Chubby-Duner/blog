```js
let arr = [
  {
    name: '小猫咪',
    age: '18'
  },
  {
    name: '阿花',
    age: '18'
  }
]
let obj = {}
for (let i = 0; i < arr.length; i ++) {
  // 如果是数字的话转一下就会带上引号
  obj[JSON.stringify(arr[i].name)] = arr[i].age
}
console.log(obj);
```

::: warning 重点
obj[b[item].uuid] = b[item].name
:::
