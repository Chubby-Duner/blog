```js
alertMessage(text, cla) {
  let arr = {
    'success': '成功',
    'error': '失败'
  };
  let texts = arr[cla];
  this.$notify({
    title: texts,
    message: text,
    type: cla,
    positon: 'bottom-right',
    duration: 2500
  })
}

// 使用
this.alertMessage('请求成功！', 'error')
```
